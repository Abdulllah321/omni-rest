    import jwt from 'jsonwebtoken';
    import bcrypt from 'bcryptjs';
    import { Request, Response, NextFunction } from 'express';

    const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

    export interface AuthUser {
      id: number;
      email: string;
      role: string;
    }

    declare global {
      namespace Express {
        interface Request {
          user?: AuthUser;
        }
      }
    }

    export const authenticate = async (req: Request, res: Response, next: NextFunction) => {
      try {
        const token = req.headers.authorization?.split(' ')[1];

        if (!token) {
          return res.status(401).json({ error: 'No token provided' });
        }

        const decoded = jwt.verify(token, JWT_SECRET) as AuthUser;
        req.user = decoded;
        next();
      } catch (error) {
        res.status(401).json({ error: 'Invalid token' });
      }
    };

    export const requireRole = (roles: string[]) => {
      return (req: Request, res: Response, next: NextFunction) => {
        if (!req.user || !roles.includes(req.user.role)) {
          return res.status(403).json({ error: 'Insufficient permissions' });
        }
        next();
      };
    };

    import { PrismaClient } from '@prisma/client';
    import { AuthUser } from './auth';

    const prisma = new PrismaClient();

    export const createGuards = (currentUser?: AuthUser) => ({
      product: {
        POST: async () => {
          if (!currentUser || currentUser.role !== 'admin') {
            return 'Only admins can create products';
          }
        },
        DELETE: async ({ id }: { id: number }) => {
          if (!currentUser || currentUser.role !== 'admin') {
            return 'Only admins can delete products';
          }
          const orderCount = await prisma.order.count({
            where: {
              items: { some: { productId: id } },
              status: { in: ['pending', 'processing'] }
            }
          });
          if (orderCount > 0) return 'Cannot delete products with active orders';
        },
      },
      order: {
        POST: async () => !currentUser ? 'Authentication required' : undefined,
        PATCH: async ({ id }: { id: number }) => {
          if (!currentUser) return 'Authentication required';
          const order = await prisma.order.findUnique({ where: { id }, select: { userId: true } });
          if (!order || order.userId !== currentUser.id) return 'Can only update your own orders';
        },
      },
    });

    ```typescript
    import express from 'express';
    import cors from 'cors';
    import helmet from 'helmet';
    import { PrismaClient } from '@prisma/client';
    import { expressAdapter, generateOpenApiSpec } from 'omni-rest/express';
    import swaggerUi from 'swagger-ui-express';
    import { authenticate } from './auth';
    import { createGuards } from './guards';

    const app = express();
    const prisma = new PrismaClient();
    const PORT = process.env.PORT || 3000;

    app.use(helmet());
    app.use(cors());
    app.use(express.json());

    const apiConfig = {
      allow: ['product', 'category', 'order'],
      guards: createGuards(),
      defaultLimit: 20,
      maxLimit: 100,
    };

    app.use('/api', authenticate, expressAdapter(prisma, apiConfig));

    const spec = generateOpenApiSpec(prisma, {
      title: 'E-commerce API',
      version: '1.0.0',
      basePath: '/api',
      allow: apiConfig.allow,
      servers: [{ url: 'http://localhost:3000', description: 'Development' }],
    });

    app.get('/openapi.json', (req, res) => res.json(spec));
    app.use('/docs', swaggerUi.serve, swaggerUi.setup(spec));

    app.listen(PORT, () => {
      console.log(`🚀 Server running on http://localhost:${PORT}`);
    });
    ```

    // hooks/useApi.ts
    import { useState, useEffect } from 'react';

    export function useProducts() {
      const [data, setData] = useState(null);
      const [loading, setLoading] = useState(true);

      useEffect(() => {
        fetch('/api/product')
          .then(res => res.json())
          .then(setData)
          .finally(() => setLoading(false));
      }, []);

      return { data, loading };
    }
