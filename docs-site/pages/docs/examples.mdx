### Authentication Middleware

```typescript
// src/auth.ts
import jwt from 'jsonwebtoken';
import bcrypt from 'bcryptjs';
import { Request, Response, NextFunction } from 'express';

const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

export interface AuthUser {
  id: number;
  email: string;
  role: string;
}

declare global {
  namespace Express {
    interface Request {
      user?: AuthUser;
    }
  }
}

export const authenticate = async (req: Request, res: Response, next: NextFunction) => {
  try {
    const token = req.headers.authorization?.split(' ')[1];

    if (!token) {
      return res.status(401).json({ error: 'No token provided' });
    }

    const decoded = jwt.verify(token, JWT_SECRET) as AuthUser;
    req.user = decoded;
    next();
  } catch (error) {
    res.status(401).json({ error: 'Invalid token' });
  }
};

export const requireRole = (roles: string[]) => {
  return (req: Request, res: Response, next: NextFunction) => {
    if (!req.user) {
      return res.status(401).json({ error: 'Authentication required' });
    }

    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }

    next();
  };
};

export const hashPassword = async (password: string): Promise<string> => {
  return bcrypt.hash(password, 12);
};

export const verifyPassword = async (password: string, hash: string): Promise<boolean> => {
  return bcrypt.compare(password, hash);
};

export const generateToken = (user: AuthUser): string => {
  return jwt.sign(user, JWT_SECRET, { expiresIn: '7d' });
};
```

### Guards Configuration

```typescript
// src/guards.ts
import { PrismaClient } from '@prisma/client';
import { AuthUser } from './auth';

const prisma = new PrismaClient();

export const createGuards = (currentUser?: AuthUser) => ({
  product: {
    POST: async () => {
      if (!currentUser || currentUser.role !== 'admin') {
        return 'Only admins can create products';
      }
    },
    DELETE: async ({ id }: { id: number }) => {
      if (!currentUser || currentUser.role !== 'admin') {
        return 'Only admins can delete products';
      }

      // Check if product has active orders
      const orderCount = await prisma.order.count({
        where: {
          items: { some: { productId: id } },
          status: { in: ['pending', 'processing'] }
        }
      });

      if (orderCount > 0) {
        return 'Cannot delete products with active orders';
      }
    },
  },
  category: {
    DELETE: async ({ id }: { id: number }) => {
      if (!currentUser || currentUser.role !== 'admin') {
        return 'Only admins can delete categories';
      }

      // Check if category has products
      const productCount = await prisma.product.count({
        where: { categoryId: id }
      });

      if (productCount > 0) {
        return 'Cannot delete categories with products';
      }
    },
  },
  order: {
    POST: async () => {
      if (!currentUser) {
        return 'Authentication required to create orders';
      }
    },
    PATCH: async ({ id }: { id: number }) => {
      if (!currentUser) return 'Authentication required';

      // Users can only update their own orders
      const order = await prisma.order.findUnique({
        where: { id },
        select: { userId: true }
      });

      if (!order || order.userId !== currentUser.id) {
        return 'Can only update your own orders';
      }
    },
  },
});
```

### Server Setup

```typescript
// src/server.ts
import express from 'express';
import cors from 'cors';
import helmet from 'helmet';
import { PrismaClient } from '@prisma/client';
import { expressAdapter, generateOpenApiSpec } from 'omni-rest/express';
import swaggerUi from 'swagger-ui-express';
import { authenticate } from './auth';
import { createGuards } from './guards';

const app = express();
const prisma = new PrismaClient();
const PORT = process.env.PORT || 3000;

// Middleware
app.use(helmet());
app.use(cors());
app.use(express.json());

// Health check
app.get('/health', (req, res) => {
  res.json({ status: 'OK', timestamp: new Date().toISOString() });
});

// API Configuration
const apiConfig = {
  allow: ['product', 'category', 'order'],

  guards: createGuards(),

  beforeOperation: async ({ model, method, id, body }) => {
    console.log(`[${new Date().toISOString()}] ${method} ${model}${id ? `/${id}` : ''}`, {
      userId: req.user?.id,
      body: method !== 'GET' ? body : undefined,
    });
  },

  afterOperation: async ({ model, method, result }) => {
    if (method === 'POST' && model === 'order') {
      // Update product inventory
      const order = result as any;
      for (const item of order.items) {
        await prisma.product.update({
          where: { id: item.productId },
          data: { inventory: { decrement: item.quantity } }
        });
      }
    }
  },

  defaultLimit: 20,
  maxLimit: 100,
};

// Mount API with authentication
app.use('/api', authenticate, expressAdapter(prisma, apiConfig));

// OpenAPI Documentation
const spec = generateOpenApiSpec(prisma, {
  title: 'E-commerce API',
  version: '1.0.0',
  basePath: '/api',
  allow: apiConfig.allow,
  servers: [
    { url: 'http://localhost:3000', description: 'Development' },
    { url: 'https://api.example.com', description: 'Production' },
  ],
});

app.get('/openapi.json', (req, res) => res.json(spec));
app.use('/docs', swaggerUi.serve, swaggerUi.setup(spec, {
  swaggerOptions: {
    persistAuthorization: true,
  },
}));

// Error handling
app.use((err: any, req: express.Request, res: express.Response, next: express.NextFunction) => {
  console.error(err);
  res.status(500).json({ error: 'Internal server error' });
});

// Start server
app.listen(PORT, () => {
  console.log(`🚀 Server running on http://localhost:${PORT}`);
  console.log(`📖 API docs at http://localhost:${PORT}/docs`);
  console.log(`🔗 API at http://localhost:${PORT}/api`);
});
```

### React Hook

```typescript
// hooks/useApi.ts
import { useState, useEffect } from 'react';

interface ApiResponse<T> {
  data: T;
  loading: boolean;
  error: string | null;
}

export function useApi<T>(url: string, options?: RequestInit): ApiResponse<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetch(url, options)
      .then(res => res.json())
      .then(setData)
      .catch(err => setError(err.message))
      .finally(() => setLoading(false));
  }, [url]);

  return { data, loading, error };
}

export function useProducts() {
  return useApi('/api/product');
}

export function useProduct(id: string) {
  return useApi(`/api/product/${id}`);
}
```

### Product Form Component

```typescript
// components/ProductForm.tsx
import { useState } from 'react';

export default function ProductForm({ onSubmit }: { onSubmit: (data: any) => void }) {
  const [formData, setFormData] = useState({
    name: '',
    description: '',
    price: '',
    inventory: '',
    categoryId: ''
  });

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    onSubmit({
      ...formData,
      price: parseFloat(formData.price),
      inventory: parseInt(formData.inventory),
      categoryId: parseInt(formData.categoryId)
    });
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <div>
        <label className="block text-sm font-medium">Name</label>
        <input
          type="text"
          value={formData.name}
          onChange={e => setFormData({...formData, name: e.target.value})}
          className="mt-1 block w-full border rounded-md px-3 py-2"
          required
        />
      </div>

      <div>
        <label className="block text-sm font-medium">Description</label>
        <textarea
          value={formData.description}
          onChange={e => setFormData({...formData, description: e.target.value})}
          className="mt-1 block w-full border rounded-md px-3 py-2"
          rows={3}
        />
      </div>

      <div className="grid grid-cols-2 gap-4">
        <div>
          <label className="block text-sm font-medium">Price</label>

            type="number"
```

## Fastify Example

A high-performance Fastify server with custom validation.

### Server Setup

```typescript
// server.ts
import Fastify from 'fastify';
import { PrismaClient } from '@prisma/client';
import { fastifyAdapter } from 'omni-rest/fastify';

const app = Fastify({ logger: true });
const prisma = new PrismaClient();

const start = async () => {
  try {
    // Register omni-rest
    await fastifyAdapter(app, prisma, {
      prefix: '/api',
      allow: ['user', 'post', 'comment'],

      guards: {
        post: {
          POST: async ({ body }) => {
            // Custom validation
            if (body.title?.length < 5) {
              return 'Title must be at least 5 characters';
            }
          },
        },
        comment: {
          POST: async ({ body }) => {
            // Check if post exists
            const post = await prisma.post.findUnique({
              where: { id: body.postId }
            });
            if (!post) {
              return 'Post not found';
            }
          },
        },
      },

      beforeOperation: async ({ model, method }) => {
        app.log.info(`${method} ${model}`);
      },
    });

    await app.listen({ port: 3000 });
    console.log('🚀 Server running on http://localhost:3000');
  } catch (err) {
    app.log.error(err);
    process.exit(1);
  }
};

start();
```
