# Getting Started

Get up and running with Omni Rest in 5 minutes.

## Prerequisites

- Node.js 18+
- A Prisma project

<Steps>

### Installation

Install the required packages using your preferred package manager.

<Tabs items={["npm", "pnpm", "yarn"]}>
  <Tabs.Tab>
    ```bash npm install omni-rest @prisma/client npm install -D prisma ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```bash pnpm add omni-rest @prisma/client pnpm add -D prisma ```
  </Tabs.Tab>
  <Tabs.Tab>
    ```bash yarn add omni-rest @prisma/client yarn add -D prisma ```
  </Tabs.Tab>
</Tabs>

### Define Your Schema

Create a `prisma/schema.prisma` file with your models.

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = "file:./dev.db"
}

model Product {
  id        Int      @id @default(autoincrement())
  name      String   @unique
  price     Float
  inventory Int      @default(0)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Category {
  id    Int     @id @default(autoincrement())
  name  String  @unique
  products Product[]
}
```

### Generate Prisma Client

<Callout type="info">
  You must regenerate the Prisma client every time you modify your schema.
</Callout>

```bash
npx prisma generate
```

    // server.ts
    import express from "express";
    import { PrismaClient } from "@prisma/client";
    import { expressAdapter } from "omni-rest/express";

    const app = express();
    const prisma = new PrismaClient();

    app.use(express.json());
    app.use("/api", expressAdapter(prisma));

    import { koaAdapter } from "omni-rest/koa";

    const app = new Koa();
    const router = new Router({ prefix: "/api" });
    const prisma = new PrismaClient();

    app.use(bodyParser());
    koaAdapter(router, prisma);

    app.use(router.routes());

    // server.ts
    import Hapi from "@hapi/hapi";
    import { PrismaClient } from "@prisma/client";
    import { hapiAdapter } from "omni-rest/hapi";

    const init = async () => {
      const prisma = new PrismaClient();
      const server = Hapi.server({ port: 3000 });

      await server.register({
        plugin: hapiAdapter,
        options: {
          prisma,
          prefix: "/api",
        }
      });

      await server.start();
      console.log("🚀 Server running on %s", server.info.uri);
    };
